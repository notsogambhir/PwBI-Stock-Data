function loadFile(file) {
  if (!file) return;
  document.getElementById('loading').classList.add('show');
  const reader = new FileReader();
  reader.onload = e => {
    try {
      const wb = XLSX.read(e.target.result, { type: 'array', cellDates: true });

      // Stock Data sheet
      const ws1 = wb.Sheets['Stock Data'];
      if (!ws1) throw new Error('Sheet "Stock Data" not found');
      const raw = XLSX.utils.sheet_to_json(ws1, { raw: false, defval: '' });
      RAW = raw.map(r => ({
        Datetime:    r['Datetime'] || r['Date'] || '',
        Open:        +r['Open']   || 0,
        High:        +r['High']   || 0,
        Low:         +r['Low']    || 0,
        Close:       +r['Close']  || 0,
        Volume:      +r['Volume'] || 0,
        Return:      +r['Return'] || 0,
        Change:      +r['Change'] || 0,
        MA20:        +r['MA20']   || null,
        MA50:        +r['MA50']   || null,
        MA200:       +r['MA200']  || null,
        Volatility:  +r['Volatility'] || null,
        RSI:         +r['RSI']    || null,
        BB_Upper:    +r['BB_Upper'] || null,
        BB_Lower:    +r['BB_Lower'] || null,
        VWAP:        +r['VWAP']   || null,
        ATR:         +r['ATR']    || null,
        MACD:        +r['MACD']   || null,
        MACD_Signal: +r['MACD_Signal'] || null,
        Drawdown:    +r['Drawdown'] || null,
        _dt:         new Date(r['Datetime'] || r['Date'])
      })).filter(r => r.Close > 0);

      // Monthly Returns sheet
      const ws2 = wb.Sheets['Monthly Returns'];
      if (ws2) {
        MONTHLY = XLSX.utils.sheet_to_json(ws2, { raw: false, defval: '' }).map(r => ({
          Month:    r['Month'] || '',
          Open:     +r['Open']  || 0,
          High:     +r['High']  || 0,
          Low:      +r['Low']   || 0,
          Close:    +r['Close'] || 0,
          Volume:   +r['Volume'] || 0,
          Return:   +r['Return_%'] || 0
        }));
      }

      // Dashboard sheet
      const ws3 = wb.Sheets['Dashboard'];
      if (ws3) {
        XLSX.utils.sheet_to_json(ws3, { raw: false }).forEach(r => {
          if (r['Metric'] && r['Value'] !== undefined) DASH[r['Metric']] = r['Value'];
        });
      }

      initDashboard();
    } catch(err) {
      alert('Error reading file: ' + err.message);
      document.getElementById('loading').classList.remove('show');
    }
  };
  reader.readAsArrayBuffer(file);
}

function clr(v, t=0) { return v > t ? 'var(--green)' : v < t ? 'var(--red)' : 'var(--muted)' }
function fmtN(v, d=2) { return v == null || isNaN(v) ? 'N/A' : (+v).toFixed(d) }
function fmtK(v) {
  if (v >= 1e6) return (v/1e6).toFixed(1)+'M';
  if (v >= 1e3) return (v/1e3).toFixed(0)+'K';
  return v;
}
function pct(v) { return (v>=0?'+':'')+fmtN(v,2)+'%' }
function last(arr, key) { const v = arr.filter(r=>r[key]!=null&&!isNaN(r[key])); return v.length?v[v.length-1][key]:null }
function destroyChart(id) { if (charts[id]) { charts[id].destroy(); delete charts[id]; } }

function renderKPI1() {
  if (!FILTERED.length) return;
  const d = FILTERED;
  const latest = d[d.length-1];
  const prev   = d[d.length-2] || latest;

  const yearStart = new Date(latest._dt.getFullYear(), 0, 1);
  const ytdFirst  = d.find(r => r._dt >= yearStart);
  const ytd       = ytdFirst ? (latest.Close - ytdFirst.Close) / ytdFirst.Close * 100 : 0;
  const hi52      = Math.max(...d.slice(-Math.min(252*7,d.length)).map(r=>r.High));
  const lo52      = Math.min(...d.slice(-Math.min(252*7,d.length)).map(r=>r.Low));
  const rsi       = last(d, 'RSI');
  const dayChgPct = prev.Close ? (latest.Close - prev.Close) / prev.Close * 100 : 0;

  const rsiSignal = rsi >= 70 ? '⚠ Overbought' : rsi <= 30 ? '✅ Oversold' : rsi >= 55 ? 'Bullish' : rsi <= 45 ? 'Bearish' : 'Neutral';
  const rsiClass  = rsi >= 70 ? 'c-red' : rsi <= 30 ? 'c-green' : 'c-amber';

  // Build spark for CMP card
  const sparkPts = d.slice(-40).map(r=>r.Close);
  const sparkMin = Math.min(...sparkPts), sparkMax = Math.max(...sparkPts);
  
  const pathD = sparkPts.map((v,i)=>{
    const len = sparkPts.length > 1 ? sparkPts.length - 1 : 1;
    const x = i/len*100;
    const y = 22 - (v-sparkMin)/(sparkMax-sparkMin||1)*20;
    return (i===0?'M':'L')+x.toFixed(1)+','+y.toFixed(1);
  }).join(' ');

  document.getElementById('kpi-row-1').innerHTML = `
    <div class="kpi-card c-blue">
      <div class="kpi-label">Current Price</div>
      <div class="kpi-value v-blue">₹${fmtN(latest.Close,2)}</div>
      <div class="kpi-sub">${pct(dayChgPct)} vs prev bar</div>
      <svg width="100%" height="24" viewBox="0 0 100 24" preserveAspectRatio="none" style="margin-top:4px">
        <path d="${pathD}" fill="none" stroke="var(--blue)" stroke-width="1.5"/>
        <circle cx="100" cy="${(22-(sparkPts[sparkPts.length-1]-sparkMin)/(sparkMax-sparkMin||1)*20).toFixed(1)}" r="2" fill="var(--blue-l)"/>
      </svg>
    </div>
    <div class="kpi-card ${dayChgPct>=0?'bg-green c-green':'bg-red c-red'}">
      <div class="kpi-label">Day Change</div>
      <div class="kpi-value ${dayChgPct>=0?'v-green':'v-red'}">${pct(dayChgPct)}</div>
      <div class="kpi-sub" style="color:${dayChgPct>=0?'var(--green)':'var(--red)'}">₹${fmtN(latest.Close-prev.Close,2)}</div>
    </div>
    <div class="kpi-card c-green">
      <div class="kpi-label">YTD Return</div>
      <div class="kpi-value ${ytd>=0?'v-green':'v-red'}">${pct(ytd)}</div>
      <div class="kpi-sub" style="color:var(--dim)">Jan 1 → Now</div>
    </div>
    <div class="kpi-card c-dim">
      <div class="kpi-label">52W High</div>
      <div class="kpi-value v-muted">₹${fmtN(hi52,2)}</div>
      <div class="kpi-sub v-red">${pct((latest.Close-hi52)/hi52*100)} from high</div>
    </div>
    <div class="kpi-card c-dim">
      <div class="kpi-label">52W Low</div>
      <div class="kpi-value v-muted">₹${fmtN(lo52,2)}</div>
      <div class="kpi-sub v-green">${pct((latest.Close-lo52)/lo52*100)} from low</div>

  const start = Math.floor(zoomL / 100 * FILTERED.length);

// HEATMAP
// ═══════════════════════════════════════════════════════════════
function renderHeatmap() {
  if (!MONTHLY.length) return;
  const container = document.getElementById('heatmap');
  const months = ['Jan','Feb','Mar','Apr','May','Jun','Jul','Aug','Sep','Oct','Nov','Dec'];
  const years  = [...new Set(MONTHLY.map(r => r.Month.split('-')[0]))].sort();

  const allRets = MONTHLY.map(r => r.Return).filter(v => !isNaN(v));
  const minRet  = Math.min(...allRets), maxRet = Math.max(...allRets);

  function hmColor(v) {
    if (isNaN(v) || v == null) return 'rgba(30,45,69,0.3)';
    const norm = Math.max(-1, Math.min(1, v / 8));
    if (norm >= 0) return `rgba(16,185,129,${0.1 + norm * 0.8})`;
    return `rgba(239,68,68,${0.1 + Math.abs(norm) * 0.8})`;
  }
  function textColor(v) {
    if (isNaN(v)||v==null) return 'var(--dim)';
    const abs = Math.abs(v);
    if (abs > 4) return '#fff';
    return abs > 2 ? 'rgba(255,255,255,0.8)' : 'var(--muted)';
  }

  container.style.gridTemplateColumns = `40px repeat(12,1fr)`;

  let html = `<div class="hm-cell hm-header"></div>`;
  months.forEach(m => html += `<div class="hm-cell hm-header">${m}</div>`);

  years.forEach(yr => {
    html += `<div class="hm-cell hm-year">${yr}</div>`;
    months.forEach((m, mi) => {
      const key  = `${yr}-${String(mi+1).padStart(2,'0')}`;
      const row  = MONTHLY.find(r => r.Month === key);
      const ret  = row ? row.Return : null;
      const disp = ret != null ? fmtN(ret,1) : '—';
      html += `<div class="hm-cell" 
        style="background:${hmColor(ret)};color:${textColor(ret)}"
        title="${key}: ${ret!=null?pct(ret):'N/A'}"
        onclick="drillthrough('${key}')">${disp}</div>`;
    });
  });
  container.innerHTML = html;
}

function drillthrough(month) {
  // Switch to data table page and filter by month
  showTab(4);
  document.getElementById('table-search').value = month.substring(0,7);
  filterTable();
}

function setTableFilter(f) {
  tableFilter = f;
  document.querySelectorAll('.table-controls .preset-btn').forEach(b => b.classList.remove('active'));
  document.getElementById('tbl-'+f).classList.add('active');
  tablePage = 1;
  renderTable();
}

function filterTable() {
  tableSearch = document.getElementById('table-search').value.toLowerCase();
  tablePage = 1;
  renderTable();
}

let sortDir = {};
function sortTable(col) {
  if (sortDir[col] === undefined) sortDir[col] = false;
  sortDir[col] = !sortDir[col];
  tableData.sort((a,b) => {
    // Utilize internal `_dt` object representation to cleanly sort chronologically
    const av = col === 'Datetime' ? a._dt : a[col];
    const bv = col === 'Datetime' ? b._dt : b[col];
    if (av == null) return 1; if (bv == null) return -1;
    return sortDir[col] ? (av > bv ? 1 : (av < bv ? -1 : 0)) : (av < bv ? 1 : (av > bv ? -1 : 0));
  });
  renderTable();
}

function renderTable() {
  let data = [...tableData];

  // Apply filter
  if (tableFilter === 'highvol') {
    const avg = data.reduce((s,r)=>s+r.Volume,0)/data.length;
    data = data.filter(r => r.Volume > avg * 1.5);
  } else if (tableFilter === 'ob') {
    data = data.filter(r => r.RSI >= 70);
  } else if (tableFilter === 'os') {
    data = data.filter(r => r.RSI <= 30);
  }

  // Search
  if (tableSearch) {
    data = data.filter(r => {
      const dt = r.Datetime ? String(r.Datetime).toLowerCase() : '';
      return dt.includes(tableSearch) ||
             String(r.Close||'').includes(tableSearch);
    });
  }

  const total = data.length;
  const pages = Math.ceil(total / ROWS_PER_PAGE);
  const start = (tablePage - 1) * ROWS_PER_PAGE;
  const slice = data.slice(start, start + ROWS_PER_PAGE);

  // Max volume for data bars
  const maxVol = data.length ? Math.max(...data.map(r=>r.Volume||0)) : 0;
  const maxRetAbs = data.length ? Math.max(...data.map(r=>Math.abs(r.Return||0))) : 0;

  const tbody = document.getElementById('table-body');
  tbody.innerHTML = slice.map(r => {
    const retColor = r.Return >= 0 ? 'var(--green)' : 'var(--red)';
    const macdColor = (r.MACD||0) >= 0 ? 'var(--green)' : 'var(--red)';
    const rsiIcon  = r.RSI >= 70 ? '<span class="rsi-icon" style="color:var(--red)">⚠</span>' :
                     r.RSI <= 30 ? '<span class="rsi-icon" style="color:var(--green)">✓</span>' :
                                   '<span class="rsi-icon" style="color:var(--dim)">→</span>';
    const volPct = maxVol ? (r.Volume||0)/maxVol*100 : 0;
    const retPct = maxRetAbs ? Math.abs(r.Return||0)/maxRetAbs*100 : 0;
    const ddColor = (r.Drawdown||0) < -0.1 ? 'var(--red)' : 'var(--dim)';

    return `<tr>
      <td>${r.Datetime || ''}</td>
      <td>₹${fmtN(r.Open,2)}</td>
      <td>₹${fmtN(r.High,2)}</td>
      <td>₹${fmtN(r.Low,2)}</td>
      <td style="color:var(--text);font-weight:600">₹${fmtN(r.Close,2)}</td>
      <td class="bar-cell">
        <div class="data-bar" style="width:${volPct.toFixed(0)}%;background:rgba(37,99,235,0.35)"></div>
        <span style="position:relative;z-index:1">${fmtK(r.Volume||0)}</span>
      </td>
      <td class="bar-cell" style="color:${retColor}">
        <div class="data-bar" style="width:${retPct.toFixed(0)}%;background:${r.Return>=0?'rgba(16,185,129,0.35)':'rgba(239,68,68,0.35)'}"></div>
        <span style="position:relative;z-index:1">${r.Return!=null?pct(r.Return*100):'—'}</span>
      </td>
      <td style="color:${r.RSI>=70?'var(--red)':r.RSI<=30?'var(--green)':'var(--muted)'}">${rsiIcon}${fmtN(r.RSI,1)}</td>
      <td style="color:${macdColor}">${fmtN(r.MACD,3)}</td>
      <td>₹${fmtN(r.ATR,2)}</td>
      <td style="color:${ddColor}">${r.Drawdown!=null?pct((r.Drawdown||0)*100):'—'}</td>
    </tr>`;
  }).join('');

  // Pagination
  document.getElementById('table-info').textContent =
    `Showing ${total > 0 ? start+1 : 0}–${Math.min(start+ROWS_PER_PAGE,total)} of ${total.toLocaleString()} rows`;

  const btnContainer = document.getElementById('page-btns');
  let btns = '';
  const showPages = [];
  if (pages <= 7) { for(let i=1;i<=pages;i++) showPages.push(i); }
  else {
    showPages.push(1);
    if (tablePage > 3) showPages.push('…');
    for(let i=Math.max(2,tablePage-1); i<=Math.min(pages-1,tablePage+1); i++) showPages.push(i);
    if (tablePage < pages-2) showPages.push('…');
    showPages.push(pages);
  }
  showPages.forEach(p => {
    if (p === '…') btns += `<span class="page-btn" style="cursor:default;pointer-events:none">…</span>`;
    else btns += `<button class="page-btn ${p===tablePage?'active':''}" onclick="goPage(${p})">${p}</button>`;
  });
  btnContainer.innerHTML = btns;
}

function goPage(p) { tablePage = p; renderTable(); }
